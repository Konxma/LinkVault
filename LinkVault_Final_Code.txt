package com.konxma.linkvault.infrastructure;

import java.security.MessageDigest;
import java.security.NoSuchAlgorithmException;
import java.util.Base64;

public class PasswordHasher {

  // 袦械褌芯写 写谢褟 锌械褉械褌胁芯褉械薪薪褟 蟹胁懈褔邪泄薪芯谐芯 锌邪褉芯谢褟 薪邪 薪械褔懈褌邪斜械谢褜薪懈泄 褏械褕
  public static String hashPassword(String rawPassword) {
    try {
      MessageDigest digest = MessageDigest.getInstance("SHA-256");
      byte[] hash = digest.digest(rawPassword.getBytes());
      return Base64.getEncoder().encodeToString(hash);
    } catch (NoSuchAlgorithmException e) {
      throw new RuntimeException("袩芯屑懈谢泻邪 褏械褕褍胁邪薪薪褟 锌邪褉芯谢褟", e);
    }
  }

  // 袦械褌芯写 写谢褟 锌械褉械胁褨褉泻懈, 褔懈 蟹斜褨谐邪褦褌褜褋褟 胁胁械写械薪懈泄 锌邪褉芯谢褜 褨蟹 褏械褕械屑 褍 斜邪蟹褨
  public static boolean verifyPassword(String rawPassword, String hashedPassword) {
    return hashPassword(rawPassword).equals(hashedPassword);
  }
}

import com.konxma.linkvault.model.Category;
import java.sql.Connection;
import java.sql.PreparedStatement;
import java.sql.ResultSet;
import java.util.ArrayList;
import java.util.List;

public class CategoryRepository {

  // 小褌胁芯褉械薪薪褟 (Create)
  public void save(Category category) {
    String sql = "INSERT INTO categories (user_id, name, description) VALUES (?, ?, ?)";
    try (Connection conn = DatabaseConnection.getInstance().getConnection();
        PreparedStatement pstmt = conn.prepareStatement(sql)) {
      pstmt.setInt(1, category.getUserId());
      pstmt.setString(2, category.getName());
      pstmt.setString(3, category.getDescription());
      pstmt.executeUpdate();
    } catch (Exception e) {
      e.printStackTrace();
    }
  }

  // 效懈褌邪薪薪褟 (Read)
  public List<Category> findByUserId(int userId) {
    List<Category> categories = new ArrayList<>();
    String sql = "SELECT * FROM categories WHERE user_id = ?";
    try (Connection conn = DatabaseConnection.getInstance().getConnection();
        PreparedStatement pstmt = conn.prepareStatement(sql)) {
      pstmt.setInt(1, userId);
      ResultSet rs = pstmt.executeQuery();
      while (rs.next()) {
        categories.add(new Category(
            rs.getInt("category_id"),
            rs.getInt("user_id"),
            rs.getString("name"),
            rs.getString("description")
        ));
      }
    } catch (Exception e) {
      e.printStackTrace();
    }
    return categories;
  }

  // 袨薪芯胁谢械薪薪褟 (Update)
  public void update(Category category) {
    String sql = "UPDATE categories SET name = ?, description = ? WHERE category_id = ?";
    try (Connection conn = DatabaseConnection.getInstance().getConnection();
        PreparedStatement pstmt = conn.prepareStatement(sql)) {
      pstmt.setString(1, category.getName());
      pstmt.setString(2, category.getDescription());
      pstmt.setInt(3, category.getCategoryId());
      pstmt.executeUpdate();
    } catch (Exception e) {
      e.printStackTrace();
    }
  }

  // 袙懈写邪谢械薪薪褟 (Delete)
  public void delete(int id) {
    String sql = "DELETE FROM categories WHERE category_id = ?";
    try (Connection conn = DatabaseConnection.getInstance().getConnection();
        PreparedStatement pstmt = conn.prepareStatement(sql)) {
      pstmt.setInt(1, id);
      pstmt.executeUpdate();
    } catch (Exception e) {
      e.printStackTrace();
    }
  }
}

public class DatabaseConnection {

  private static final String URL = "jdbc:postgresql://localhost:5432/linkvault_db";
  private static final String USER = "postgres";
  private static final String PASSWORD = "postgres";
  private static final int POOL_SIZE = 10;

  private static DatabaseConnection instance;
  private final BlockingQueue<Connection> connectionPool;

  /**
   * 袩褉懈胁邪褌薪懈泄 泻芯薪褋褌褉褍泻褌芯褉, 褟泻懈泄 褨薪褨褑褨邪谢褨蟹褍褦 锌褍谢 蟹'褦写薪邪薪褜 蟹邪写邪薪芯谐芯 褉芯蟹屑褨褉褍.
   */
  private DatabaseConnection() {
    connectionPool = new ArrayBlockingQueue<>(POOL_SIZE);
    try {
      for (int i = 0; i < POOL_SIZE; i++) {
        connectionPool.add(DriverManager.getConnection(URL, USER, PASSWORD));
      }
      System.out.println("袩褍谢 蟹'褦写薪邪薪褜 (" + POOL_SIZE + " 褕褌.) 褍褋锌褨褕薪芯 褋褌胁芯褉械薪芯!");
    } catch (SQLException e) {
      throw new RuntimeException("袩芯屑懈谢泻邪 褋褌胁芯褉械薪薪褟 锌褍谢褍 蟹'褦写薪邪薪褜", e);
    }
  }

  /**
   * 袩芯胁械褉褌邪褦 褦写懈薪懈泄 械泻蟹械屑锌谢褟褉 泻谢邪褋褍 DatabaseConnection.
   *
   * @return 械泻蟹械屑锌谢褟褉 DatabaseConnection.
   */
  public static synchronized DatabaseConnection getInstance() {
    if (instance == null) {
      instance = new DatabaseConnection();
    }
    return instance;
  }

  /**
   * 袨褌褉懈屑褍褦 蟹'褦写薪邪薪薪褟 蟹 锌褍谢褍. 携泻褖芯 锌褍谢 锌芯褉芯卸薪褨泄, 锌芯褌褨泻 褔械泻邪褌懈屑械.

   */
  public Connection getConnection() {
    try {
      return connectionPool.take();
    } catch (InterruptedException e) {
      Thread.currentThread().interrupt();
      throw new RuntimeException("袩芯屑懈谢泻邪 芯褌褉懈屑邪薪薪褟 蟹'褦写薪邪薪薪褟", e);
    }

  /**

   * 携泻褖芯 蟹'褦写薪邪薪薪褟 蟹邪泻褉懈褌械, 褋褌胁芯褉褞褦褌褜褋褟 薪芯胁械.

   * @param connection 蟹'褦写薪邪薪薪褟, 褟泻械 锌芯褌褉褨斜薪芯 锌芯胁械褉薪褍褌懈.
   */
  public void releaseConnection(Connection connection) {
    if (connection != null) {
      try {
        if (!connection.isClosed()) {
          connectionPool.offer(connection);
        } else {
          connectionPool.offer(DriverManager.getConnection(URL, USER, PASSWORD));
        }
      } catch (SQLException e) {
        System.out.println("袩芯屑懈谢泻邪 锌芯胁械褉薪械薪薪褟 蟹'褦写薪邪薪薪褟: " + e.getMessage());
      }

  }

package com.konxma.linkvault.repository;

import java.lang.reflect.Field;
import java.sql.ResultSet;
import java.util.ArrayList;
import java.util.List;

public class EntityMapper<T> {

  private final Class<T> clazz;

  public EntityMapper(Class<T> clazz) {
    this.clazz = clazz;
  }

  public List<T> mapResultSetToList(ResultSet rs) throws Exception {
    List<T> resultList = new ArrayList<>();

    while (rs.next()) {
      // 小褌胁芯褉褞褦屑芯 薪芯胁懈泄 锌褍褋褌懈泄 芯斜'褦泻褌 锌芯褌褉褨斜薪芯谐芯 泻谢邪褋褍
      T entity = clazz.getDeclaredConstructor().newInstance();

      // 袩褉芯褏芯写懈屑芯褋褟 锌芯 胁褋褨褏 锌芯谢褟褏 褑褜芯谐芯 泻谢邪褋褍
      for (Field field : clazz.getDeclaredFields()) {
        field.setAccessible(true); // 袛芯蟹胁芯谢褟褦屑芯 蟹邪锌懈褋 褍 锌褉懈胁邪褌薪褨 锌芯谢褟
        String columnName = getColumnName(field);

        try {
          Object value = rs.getObject(columnName);
          if (value != null) {
            field.set(entity, value);
          }
        } catch (Exception e) {
          // 携泻褖芯 褌邪泻芯褩 泻芯谢芯薪泻懈 薪械屑邪褦 胁 ResultSet, 锌褉芯褋褌芯 泄写械屑芯 写邪谢褨
        }
      }
      resultList.add(entity);
    }
    return resultList;
  }

  private String getColumnName(Field field) {
    // 袗胁褌芯屑邪褌懈褔薪芯 泻芯薪胁械褉褌褍褦屑芯 camelCase 褍 snake_case (薪邪锌褉懈泻谢邪写 userId -> user_id)
    String regex = "([a-z])([A-Z]+)";
    String replacement = "$1_$2";
    return field.getName().replaceAll(regex, replacement).toLowerCase();
  }
}

      pstmt.setInt(1, linkId);
      ResultSet rs = pstmt.executeQuery();
      while (rs.next()) {
        tags.add(new Tag(rs.getInt("tag_id"), rs.getString("name")));
      }
    } catch (SQLException e) {

  public boolean deleteLink(int linkId) {

package com.konxma.linkvault.repository;

import java.sql.Connection;
import java.sql.SQLException;

public class UnitOfWork implements AutoCloseable {

  private final Connection connection;

  public UnitOfWork() throws SQLException {
    // 袨褌褉懈屑褍褦屑芯 蟹'褦写薪邪薪薪褟 褨 胁懈屑懈泻邪褦屑芯 邪胁褌芯屑邪褌懈褔薪械 蟹斜械褉械卸械薪薪褟 泻芯卸薪芯谐芯 蟹邪锌懈褌褍
    this.connection = DatabaseConnection.getInstance().getConnection();
    this.connection.setAutoCommit(false);
  }

  public Connection getConnection() {
    return connection;
  }

  public void commit() throws SQLException {
    // 袟斜械褉褨谐邪褦屑芯 胁褋褨 蟹屑褨薪懈 褉邪蟹芯屑 褦写懈薪芯褞 褌褉邪薪蟹邪泻褑褨褦褞
    connection.commit();
  }

  public void rollback() {
    try {
      if (connection != null) {
        // 袙褨写泻芯褔褍褦屑芯 胁褋褨 蟹屑褨薪懈 褍 褉邪蟹褨 斜褍写褜-褟泻芯褩 锌芯屑懈谢泻懈
        connection.rollback();
      }
    } catch (SQLException e) {
      e.printStackTrace();
    }
  }

  @Override
  public void close() throws SQLException {
    if (connection != null) {
      // 袩芯胁械褉褌邪褦屑芯 褋褌邪薪写邪褉褌薪褨 薪邪谢邪褕褌褍胁邪薪薪褟 锌械褉械写 蟹邪泻褉懈褌褌褟屑
      connection.setAutoCommit(true);
      connection.close();
    }
  }
}

package com.konxma.linkvault.repository;

import com.konxma.linkvault.model.User;

import java.sql.Connection;
import java.sql.PreparedStatement;
import java.sql.ResultSet;
import java.sql.Statement;
import java.util.List;

public class UserRepository {
  private final EntityMapper<User> userMapper = new EntityMapper<>(User.class);
  private final IdentityMap<User> identityMap = new IdentityMap<>();

  public User save(User user) {
    // 袙懈泻芯褉懈褋褌芯胁褍褦屑芯 褍薪褨胁械褉褋邪谢褜薪懈泄 SQL 斜械蟹 褋锌械褑懈褎褨褔薪芯谐芯 "RETURNING"
    String sql = "INSERT INTO users (username, email, password_hash) VALUES (?, ?, ?)";

    try (Connection conn = DatabaseConnection.getInstance().getConnection();
        // 袛芯写邪褦屑芯 胁泻邪蟹褨胁泻褍 RETURN_GENERATED_KEYS
        PreparedStatement pstmt = conn.prepareStatement(sql, Statement.RETURN_GENERATED_KEYS)) {

      pstmt.setString(1, user.getUsername());
      pstmt.setString(2, user.getEmail());
      pstmt.setString(3, user.getPasswordHash());
      pstmt.executeUpdate();

      // 袨褌褉懈屑褍褦屑芯 蟹谐械薪械褉芯胁邪薪懈泄 斜邪蟹芯褞 写邪薪懈褏 ID
      ResultSet rs = pstmt.getGeneratedKeys();
      if (rs.next()) {
        user.setUserId(rs.getInt(1)); // 1 - 褑械 薪芯屑械褉 泻芯谢芯薪泻懈 蟹 ID
        identityMap.put(user.getUserId(), user);
      }
    } catch (Exception e) {
      e.printStackTrace();
    }
    return user;
  }

  public User findById(int id) {
    User cachedUser = identityMap.get(id);
    if (cachedUser != null) {
      return cachedUser;
    }
    String sql = "SELECT * FROM users WHERE user_id = ?";
    try (Connection conn = DatabaseConnection.getInstance().getConnection();
        PreparedStatement pstmt = conn.prepareStatement(sql)) {
      pstmt.setInt(1, id);
      ResultSet rs = pstmt.executeQuery();
      List<User> users = userMapper.mapResultSetToList(rs);
      if (!users.isEmpty()) {
        User user = users.get(0);
        identityMap.put(user.getUserId(), user);
        return user;
      }
    } catch (Exception e) {
      e.printStackTrace();
    }
    return null;
  }

  public User findByEmail(String email) {
    String sql = "SELECT * FROM users WHERE email = ?";
    try (Connection conn = DatabaseConnection.getInstance().getConnection();
        PreparedStatement pstmt = conn.prepareStatement(sql)) {
      pstmt.setString(1, email);
      ResultSet rs = pstmt.executeQuery();
      List<User> users = userMapper.mapResultSetToList(rs);
      if (!users.isEmpty()) {
        User user = users.get(0);
        identityMap.put(user.getUserId(), user);
        return user;
      }
    } catch (Exception e) {
      e.printStackTrace();
    }
    return null;
  }

  public void clearCache() {
    identityMap.clear();
  }
}

package com.konxma.linkvault.service;

import com.konxma.linkvault.model.User;
import com.konxma.linkvault.model.Category;
import com.konxma.linkvault.repository.UserRepository;
import com.konxma.linkvault.repository.CategoryRepository;
import com.konxma.linkvault.repository.UnitOfWork;
import com.konxma.linkvault.infrastructure.PasswordHasher;
import com.konxma.linkvault.infrastructure.EmailSender;

public class UserService {
  private final UserRepository userRepository;
  private final CategoryRepository categoryRepository;
  private final EmailSender emailSender;

  public UserService() {
    this.userRepository = new UserRepository();
    this.categoryRepository = new CategoryRepository();
    this.emailSender = new EmailSender();
  }

  public void registerNewUser(User user, String rawPassword) throws Exception {
    // 袙懈泻芯褉懈褋褌芯胁褍褦屑芯 锌褉邪胁懈谢褜薪懈泄 try-with-resources (胁懈褉褨褕褍褦 卸芯胁褌械 锌芯锌械褉械写卸械薪薪褟)
    try (UnitOfWork uow = new UnitOfWork()) {
      try {
        // 1. 啸械褕褍褦屑芯 锌邪褉芯谢褜 袩袝袪袝袛 蟹斜械褉械卸械薪薪褟屑
        String hashedPassword = PasswordHasher.hashPassword(rawPassword);
        user.setPasswordHash(hashedPassword);

        // 2. 袟斜械褉褨谐邪褦屑芯 泻芯褉懈褋褌褍胁邪褔邪
        User savedUser = userRepository.save(user);

        // 3. 小褌胁芯褉褞褦屑芯 斜邪蟹芯胁褍 泻邪褌械谐芯褉褨褞
        Category defaultCategory = new Category();
        defaultCategory.setUserId(savedUser.getUserId());
        defaultCategory.setName("袟邪谐邪谢褜薪邪 锌邪锌泻邪");
        defaultCategory.setDescription("袦芯褩 锌械褉褕褨 锌芯褋懈谢邪薪薪褟");
        categoryRepository.save(defaultCategory);

        // 4. 袩褨写褌胁械褉写卸褍褦屑芯 褌褉邪薪蟹邪泻褑褨褞
        uow.commit();

        // 5. 袙褨写锌褉邪胁谢褟褦屑芯 谢懈褋褌 锌褨写褌胁械褉写卸械薪薪褟
        emailSender.sendConfirmationEmail(savedUser.getEmail(), savedUser.getUsername());

      } catch (Exception e) {
        uow.rollback();
        throw e;
      }
    }
  }

  public User authenticate(String email, String rawPassword) {
    User user = userRepository.findByEmail(email);
    // 袨斜'褦写薪邪谢懈 if 写谢褟 褔懈褋褌芯褌懈 泻芯写褍 (胁懈褉褨褕褍褦 卸芯胁褌械 锌芯锌械褉械写卸械薪薪褟)
    if (user != null && PasswordHasher.verifyPassword(rawPassword, user.getPasswordHash())) {
      return user;
    }
    return null;
  }
}

-- 孝邪斜谢懈褑褟 1: users (袣芯褉懈褋褌褍胁邪褔褨)
-- 孝懈锌: 袛芯胁褨写泻芯胁邪 褌邪斜谢懈褑褟 (Master table).
-- 袧芯褉屑邪谢褨蟹邪褑褨褟: 3袧肖. 校褋褨 薪械泻谢褞褔芯胁褨 邪褌褉懈斜褍褌懈 (username, email, password_hash)
-- 蟹邪谢械卸邪褌褜 褌褨谢褜泻懈 胁褨写 锌械褉胁懈薪薪芯谐芯 泻谢褞褔邪 (user_id) 褨 薪械 蟹邪谢械卸邪褌褜 芯写懈薪 胁褨写 芯写薪芯谐芯.
CREATE TABLE users (
    user_id SERIAL PRIMARY KEY,
    username VARCHAR(50) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    password_hash TEXT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 孝邪斜谢懈褑褟 2: categories (袣邪褌械谐芯褉褨褩/袩邪锌泻懈)
-- 孝懈锌: 袛芯胁褨写泻芯胁邪 褌邪斜谢懈褑褟 / 孝邪斜谢懈褑褟 蟹胁'褟蟹泻褍 1:N (胁褨写 users).
-- 袧芯褉屑邪谢褨蟹邪褑褨褟: 3袧肖. 袗褌褉懈斜褍褌懈 蟹邪谢械卸邪褌褜 谢懈褕械 胁褨写 category_id.
CREATE TABLE categories (
    category_id SERIAL PRIMARY KEY,
    user_id INTEGER REFERENCES users(user_id) ON DELETE CASCADE,
    name VARCHAR(100) NOT NULL,
    description TEXT
);

-- 孝邪斜谢懈褑褟 3: links (袩芯褋懈谢邪薪薪褟)
-- 孝懈锌: 孝褉邪薪蟹邪泻褑褨泄薪邪 褌邪斜谢懈褑褟 (Transaction table) - 蟹斜械褉褨谐邪褦 芯褋薪芯胁薪褨 锌芯写褨褩 褋懈褋褌械屑懈.
-- 袧芯褉屑邪谢褨蟹邪褑褨褟: 3袧肖. 袛邪薪褨 褋锌械褑懈褎褨褔薪褨 谢懈褕械 写谢褟 泻芯薪泻褉械褌薪芯谐芯 锌芯褋懈谢邪薪薪褟.
CREATE TABLE links (
    link_id SERIAL PRIMARY KEY,
    category_id INTEGER REFERENCES categories(category_id) ON DELETE CASCADE,
    url TEXT NOT NULL,
    title VARCHAR(200) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 孝邪斜谢懈褑褟 4: tags (孝械谐懈)
-- 孝懈锌: 小谢芯胁薪懈泻 (Dictionary table).
-- 袧芯褉屑邪谢褨蟹邪褑褨褟: 3袧肖.
CREATE TABLE tags (
    tag_id SERIAL PRIMARY KEY,
    name VARCHAR(50) UNIQUE NOT NULL
);

-- 孝邪斜谢懈褑褟 5: link_tags (袟胁'褟蟹芯泻 锌芯褋懈谢邪薪褜 褌邪 褌械谐褨胁)
-- 孝懈锌: 袩褉芯屑褨卸薪邪 褌邪斜谢懈褑褟 (Junction table). 袪械邪谢褨蟹褍褦 蟹胁'褟蟹芯泻 袘邪谐邪褌芯-写芯-袘邪谐邪褌褜芯褏 (N:M).
-- 袧芯褉屑邪谢褨蟹邪褑褨褟: 3袧肖. 袩械褉胁懈薪薪懈泄 泻谢褞褔 褦 褋泻谢邪写械薪懈屑.
CREATE TABLE link_tags (
    link_id INTEGER REFERENCES links(link_id) ON DELETE CASCADE,
    tag_id INTEGER REFERENCES tags(tag_id) ON DELETE CASCADE,
